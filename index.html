Validate in Browser


function isValid($form) {
	var id = $form.find('[name=id]').val(),
			name = $form.find('[name=name]').val();
	return id.length && id.match('/^[a-z_-0-9]+/$') && name.length;

}
function onSubmit() {
	var $form = $(this);
	if (isValid($form)) {
		sendToServer($form);
	}
}

Validate on Server



function isValid(data) {
	return data.id 
			&& data.id.length 
			&& data.id.match('/^[a-z_-0-9]+/$') 
			&& data.name 
			&& data.name.length;

}
function chatAdd(req, res) {
	if (isValid(req.body)) {
		return saveToDb(
			normalize(req.body), 
			renderResponse(res)
